package com.example.mainactivity

import android.os.Bundle
import android.widget.ArrayAdapter
import android.widget.AutoCompleteTextView
import android.widget.Spinner
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val autoCountries = findViewById<AutoCompleteTextView>(R.id.autoCountries)
        val countries =
            arrayOf("Jacob", "Jhoana", "Jazer", "Josan")
        val countryAdapter = ArrayAdapter(this, android.R.layout.simple_list_item_1, countries)
        autoCountries.setAdapter(countryAdapter)

        val spinner = findViewById<Spinner>(R.id.spinnerOptions)
        val options = arrayOf("Option 1", "Option 2", "Option 3")
        val spinnerAdapter = ArrayAdapter(this, android.R.layout.simple_spinner_item, options)
        spinnerAdapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item)
        spinner.setAdapter(spinnerAdapter)
    }
}
