package com.example.mainactivity

import android.os.Bundle
import android.view.View
import android.widget.Button
import android.widget.ImageButton
import android.widget.Toast
import android.widget.ToggleButton
import androidx.appcompat.app.AppCompatActivity


class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val btnNormal = findViewById<Button?>(R.id.btnNormal)
        val btnImage = findViewById<ImageButton?>(R.id.btnImage)
        val btnToggle = findViewById<ToggleButton?>(R.id.btnToggle)

        btnNormal.setOnClickListener(View.OnClickListener { v: View? ->
            Toast.makeText(
                this,
                "Normal Button Clicked!",
                Toast.LENGTH_SHORT
            ).show()
        }
        )

        btnImage.setOnClickListener(View.OnClickListener { v: View? ->
            Toast.makeText(
                this,
                "Image Button Clicked!",
                Toast.LENGTH_SHORT
            ).show()
        }
        )

        btnToggle.setOnClickListener(View.OnClickListener { v: View? ->
            if (btnToggle.isChecked()) Toast.makeText(this, "Toggle ON", Toast.LENGTH_SHORT).show()
            else Toast.makeText(this, "Toggle OFF", Toast.LENGTH_SHORT).show()
        })
    }
}
